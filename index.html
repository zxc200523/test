function startZXingScan() { try { // 1. 簡化初始化：不傳入額外參數，預設就會開啟所有常見 1D/2D 條碼的支援 zxingReader = new window.ZXingBrowser.BrowserMultiFormatReader(); const video = document.getElementById('camera-video'); scanActive = true; // 2. ✅ 使用新版正確的 API：decodeFromVideoElement // 只要有傳入 callback (第二個參數)，它就會自動進入連續掃描的迴圈 zxingReader.decodeFromVideoElement(video, (result, err) => { if (!scanActive) return; if (result) { const barcode = result.getText().trim(); onBarcodeScanned(barcode); } // err 代表「這一個影格沒找到條碼」，這會一直瘋狂觸發，是正常現象，直接忽略即可 }); } catch(e) { console.error('ZXing 啟動失敗:', e); loadJsQRFallback(); } }
